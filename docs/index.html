Basic Operations

Create, manipulate, and format datetime instances with ease

package main

import (
    "fmt"
    "github.com/coredds/ChronoGo"
)

func main() {
    // Create datetime instances
    now := ChronoGo.Now()
    today := ChronoGo.Today()
    
    // Fluent API for date manipulation
    future := now.AddDays(7).AddHours(3).AddMinutes(30)
    
    // Format output
    fmt.Println("Now:", now.Format("2006-01-02 15:04:05"))
    fmt.Println("Future:", future.Format("Monday, January 2, 2006"))
    fmt.Println("Human:", future.DiffForHumans(now))
}

Timezone Conversions

Handle multiple timezones with proper DST support

package main

import (
    "fmt"
    "github.com/coredds/ChronoGo"
)

func main() {
    // Create in UTC
    utc := ChronoGo.UTC(2023, 12, 25, 15, 30, 0)
    
    // Convert to different timezones
    ny := utc.InTimezone("America/New_York")
    tokyo := utc.InTimezone("Asia/Tokyo")
    london := utc.InTimezone("Europe/London")
    
    fmt.Println("UTC:    ", utc.Format("15:04 MST"))
    fmt.Println("New York:", ny.Format("15:04 MST"))
    fmt.Println("Tokyo:  ", tokyo.Format("15:04 MST"))
    fmt.Println("London: ", london.Format("15:04 MST"))
    
    // Check if DST is active
    fmt.Println("DST in NY:", ny.IsDST())
}

Business Date Operations

Calculate business days and handle holidays effortlessly

package main

import (
    "fmt"
    "github.com/coredds/ChronoGo"
)

func main() {
    today := ChronoGo.Today()
    
    // Business day calculations
    workday := today.AddBusinessDays(5)
    deadline := today.SubtractBusinessDays(3)
    
    // Holiday checking
    christmas := ChronoGo.Date(2023, 12, 25)
    isHoliday := christmas.IsHoliday()
    
    // Count business days in range
    start := ChronoGo.Date(2023, 12, 1)
    end := ChronoGo.Date(2023, 12, 31)
    businessDays := start.BusinessDaysBetween(end)
    
    fmt.Println("Next workday:", workday.Format("Monday, Jan 2"))
    fmt.Println("Deadline:", deadline.Format("2006-01-02"))
    fmt.Println("Christmas is holiday:", isHoliday)
    fmt.Println("Business days in Dec:", businessDays)
}

Period Iteration

Iterate over time periods with flexible step sizes

package main

import (
    "fmt"
    "github.com/coredds/ChronoGo"
)

func main() {
    start := ChronoGo.Now()
    end := start.AddDays(7)
    
    // Create a period
    period := ChronoGo.NewPeriod(start, end)
    
    // Iterate by days
    fmt.Println("Daily schedule:")
    for _, day := range period.Days() {
        fmt.Printf("  %s\n", day.Format("Monday, Jan 2"))
    }
    
    // Iterate by hours with step
    fmt.Println("\nHourly (every 6 hours):")
    for _, hour := range period.RangeByUnitSlice(ChronoGo.Hour, 6) {
        fmt.Printf("  %s\n", hour.Format("Jan 2 15:04"))
    }
    
    // Fast range for large periods
    dates := period.FastRangeDays(1)
    fmt.Printf("Generated %d dates\n", len(dates))
}

Parsing and Serialization

Parse various formats and integrate with JSON/SQL

package main

import (
    "encoding/json"
    "fmt"
    "github.com/coredds/ChronoGo"
)

func main() {
    // Parse various formats
    dt1, _ := ChronoGo.Parse("2023-12-25T15:30:45Z")
    dt2, _ := ChronoGo.Parse("2023/12/25 15:30")
    dt3, _ := ChronoGo.Parse("20231225")
    
    // Optimized parsing
    dt4, _ := ChronoGo.ParseOptimized("1640995200")
    
    fmt.Println("ISO:", dt1.Format("2006-01-02T15:04:05Z"))
    fmt.Println("Slash:", dt2.Format("2006-01-02 15:04"))
    fmt.Println("Compact:", dt3.Format("2006-01-02"))
    fmt.Println("Unix:", dt4.Format("2006-01-02 15:04:05"))
    
    // JSON serialization
    data, _ := json.Marshal(dt1)
    fmt.Println("JSON:", string(data))
    
    // Parse from JSON
    var parsed ChronoGo.DateTime
    json.Unmarshal(data, &parsed)
    fmt.Println("Parsed:", parsed.Format("2006-01-02"))
}

Human-Readable Output

Generate user-friendly time descriptions

package main

import (
    "fmt"
    "github.com/coredds/ChronoGo"
)

func main() {
    now := ChronoGo.Now()
    
    // Different time references
    past := now.SubtractHours(2)
    future := now.AddDays(3)
    yesterday := now.SubtractDays(1)
    
    // Human-readable differences
    fmt.Println("2 hours ago:", past.DiffForHumans(now))
    fmt.Println("In 3 days:", future.DiffForHumans(now))
    fmt.Println("Yesterday:", yesterday.DiffForHumans(now))
    
    // Age calculations
    birthday := ChronoGo.Date(1990, 5, 15)
    age := birthday.Age()
    fmt.Printf("Age: %d years\n", age)
    
    // Time from now
    fmt.Println("Time from now:", future.TimeFromNow())
    fmt.Println("Time ago:", past.TimeAgo())
}
